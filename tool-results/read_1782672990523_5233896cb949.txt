   826→// Page switching with GSAP animations
   827→function switchPage(pageName) {
   828→  const pages = document.querySelectorAll('.page');
   829→  const targetPage = document.getElementById('page-' + pageName);
   830→  if (!targetPage || targetPage.classList.contains('active')) return;
   831→
   832→  // Update nav (both mobile and desktop)
   833→  document.querySelectorAll('[data-nav]').forEach(el => {
   834→    el.classList.remove('active');
   835→    // Reset sidebar link styling
   836→    if (el.classList.contains('sidebar-link')) {
   837→      el.classList.remove('text-gray-900', 'bg-pharmacy-50');
   838→      el.classList.add('text-gray-500');
   839→    }
   840→  });
   841→  
   842→  const navLinks = document.querySelectorAll('[data-nav="' + pageName + '"]');
   843→  navLinks.forEach(el => {
   844→    el.classList.add('active');
   845→    if (el.classList.contains('sidebar-link')) {
   846→      el.classList.remove('text-gray-500');
   847→      el.classList.add('text-gray-900', 'bg-pharmacy-50');
   848→    }
   849→  });
   850→
   851→  // Hide all pages
   852→  pages.forEach(p => {
   853→    if (p.classList.contains('active')) {
   854→      gsap.to(p, { 
   855→        opacity: 0, 
   856→        y: 12, 
   857→        duration: 0.25, 
   858→        ease: 'power2.in',
   859→        onComplete: () => {
   860→          p.classList.remove('active');
   861→          p.style.display = 'none';
   862→          
   863→          // Show target page
   864→          targetPage.classList.add('active');
   865→          targetPage.style.display = 'block';
   866→          gsap.fromTo(targetPage, 
   867→            { opacity: 0, y: 20 }, 
   868→            { opacity: 1, y: 0, duration: 0.4, ease: 'power2.out' }
   869→          );
   870→        }
   871→      });
   872→    }
   873→  });
   874→
   875→  // Animate cards sequentially
   876→  setTimeout(() => {
   877→    gsap.fromTo(targetPage.querySelectorAll('.card-hover, .grid > *'), 
   878→      { opacity: 0, y: 16 },
   879→      { opacity: 1, y: 0, duration: 0.35, stagger: 0.04, ease: 'power2.out' }
   880→    );
   881→  }, 300);
   882→
   883→  // Update page title
   884→  const titles = {
   885→    dashboard: 'ড্যাশবোর্ড',
   886→    inventory: 'ইনভেন্টরি',
   887→    reports: 'রিপোর্ট',
   888→    more: 'আরও'
   889→  };
   890→  const titleEl = document.querySelector('.page-title');
   891→  if (titleEl) titleEl.textContent = titles[pageName] || pageName;
   892→}
   893→
   894→// Initial page animation
   895→window.addEventListener('DOMContentLoaded', () => {
   896→  gsap.fromTo('#page-dashboard .card-hover, #page-dashboard .grid > *',
   897→    { opacity: 0, y: 20 },
   898→    { opacity: 1, y: 0, duration: 0.4, stagger: 0.05, ease: 'power2.out' }
   899→  );

   901→  // Animate welcome banner independently
