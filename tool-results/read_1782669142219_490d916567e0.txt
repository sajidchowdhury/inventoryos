     1→// ── InventoryOS: SQL Query Router (Gap 6 — Fast deterministic AI shortcuts) ──
     2→//
     3→// The SQL router intercepts natural-language business questions and answers
     4→// them directly with a Prisma query + a pre-formatted markdown response,
     5→// WITHOUT calling the LLM. This saves tokens, latency, and money for the
     6→// 20 most common questions operators ask.
     7→//
     8→// If routeQuery() returns { handled: true }, the AI chat handler should
     9→// just stream back the response verbatim and skip the LLM call entirely.
    10→// If it returns { handled: false }, the message goes to the LLM as usual.
    11→//
    12→// All currency values are formatted as BDT: ৳1234.56
    13→
    14→import { db } from "./db";
    15→
    16→// ── Types ──
    17→export interface RouteResult {
    18→  handled: boolean;
    19→  pattern?: string;
    20→  response?: string;
    21→  queryMs?: number;
    22→}
    23→
    24→export interface PatternDescriptor {
    25→  name: string;
    26→  description: string;
    27→  keywords: string[];
    28→  matchMode: "all" | "any";
    29→  excludeKeywords?: string[];
    30→}
    31→
    32→interface Pattern extends PatternDescriptor {
    33→  handler: (businessId: string, message: string) => Promise<string>;
    34→}
    35→
    36→// ── Helpers ──
    37→
    38→/** Lowercase, strip punctuation, collapse whitespace. */
    39→function normalizeMessage(s: string): string {
    40→  return s
    41→    .toLowerCase()
    42→    .replace(/[^\w\s]/g, " ")
    43→    .replace(/\s+/g, " ")
    44→    .trim();
    45→}
    46→
    47→/** Format a number as BDT currency: ৳1234.56 */
    48→function bdt(amount: number): string {
    49→  return `৳${amount.toFixed(2)}`;
    50→}
    51→
    52→/** Format a number with thousands separators. */
    53→function fmtNum(n: number): string {
    54→  return n.toLocaleString("en-US");
    55→}
    56→
    57→/** Returns midnight UTC today (start of day in UTC). */
    58→function startOfTodayUTC(): Date {
    59→  const d = new Date();
    60→  return new Date(Date.UTC(d.getUTCFullYear(), d.getUTCMonth(), d.getUTCDate()));
    61→}
    62→
    63→/** Returns midnight UTC for the first day of the current month. */
    64→function startOfMonthUTC(): Date {
    65→  const d = new Date();
    66→  return new Date(Date.UTC(d.getUTCFullYear(), d.getUTCMonth(), 1));
    67→}
    68→
    69→/** Format a Date for display (YYYY-MM-DD). */
    70→function fmtDate(d: Date): string {
    71→  return d.toISOString().slice(0, 10);
    72→}
    73→
    74→/**
    75→ * Match a normalized message against a pattern.
    76→ * - excludeKeywords checked first: if ANY present → no match
    77→ * - matchMode "all": every keyword must be present in the message
    78→ * - matchMode "any": at least one keyword must be present
    79→ */
    80→function matchesPattern(
    81→  normalizedMessage: string,
    82→  pattern: { keywords: string[]; matchMode: "all" | "any"; excludeKeywords?: string[] }
    83→): boolean {
    84→  // Check exclusions first
    85→  if (pattern.excludeKeywords && pattern.excludeKeywords.length > 0) {
    86→    for (const ex of pattern.excludeKeywords) {
    87→      const exN = normalizeMessage(ex);
    88→      if (exN && normalizedMessage.includes(exN)) return false;
    89→    }
    90→  }
    91→
    92→  // Normalize keywords too so apostrophes/punctuation in keywords match the
    93→  // already-punctuation-stripped message.
    94→  const keywords = pattern.keywords.map((k) => normalizeMessage(k)).filter(Boolean);
    95→
    96→  if (pattern.matchMode === "all") {
    97→    return keywords.every((k) => normalizedMessage.includes(k));
    98→  }
    99→  return keywords.some((k) => normalizedMessage.includes(k));
   100→}

   698→// ── Public: route a message ──
   699→export async function routeQuery(
   700→  businessId: string,
   701→  message: string
   702→): Promise<RouteResult> {
   703→  if (!message || !message.trim()) {
   704→    return { handled: false };
   705→  }
   706→  const normalized = normalizeMessage(message);
   707→
   708→  for (const pattern of PATTERNS) {
   709→    if (!matchesPattern(normalized, pattern)) continue;
   710→
   711→    const startMs = Date.now();
   712→    try {
   713→      const response = await pattern.handler(businessId, message);
   714→      const queryMs = Date.now() - startMs;
   715→      return {
   716→        handled: true,
   717→        pattern: pattern.name,
   718→        response,
   719→        queryMs,
   720→      };
   721→    } catch (err) {
   722→      const msg = err instanceof Error ? err.message : String(err);
   723→      // eslint-disable-next-line no-console
   724→      console.error(`[sql-router] pattern "${pattern.name}" threw:`, msg);
   725→      // Don't surface the error to the user — let the LLM try instead.
   726→      return { handled: false };
   727→    }
   728→  }
   729→
   730→  return { handled: false };
   731→}

   733→// ── Public: list registered patterns (for debugging / super-admin UI) ──
   734→export function getRegisteredPatterns(): PatternDescriptor[] {
   735→  return PATTERNS.map((p) => ({
   736→    name: p.name,
   737→    description: p.description,
   738→    keywords: p.keywords,
   739→    matchMode: p.matchMode,
   740→    excludeKeywords: p.excludeKeywords,
   741→  }));
   742→}
